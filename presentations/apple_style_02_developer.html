03

5分钟上手

from claw_avm import ClawSecureVMwareEngine from claw_avm.secure.engine import VMConfig # 初始化引擎 engine = ClawSecureVMwareEngine(workspace_path='./vms') # 创建安全虚拟机 config = VMConfig( name='sandbox-01', memory_mb=4096, network_isolated=True ) vm_id = engine.create_secure_vm(config)
04

核心 API

  • POST create_secure_vm(config) → vm_id
  • POST start_vm(vm_id) → bool
  • POST stop_vm(vm_id, force=False) → bool
  • GET get_vm_status(vm_id) → VMStatus
  • DELETE delete_vm(vm_id) → bool
05

快照管理

# 创建干净状态快照 snapshot = engine.create_security_snapshot( vm_id=vm_id, name='clean-state', description='分析前的基准状态' ) # 恢复（清除恶意软件） engine.restore_snapshot(vm_id, snapshot.id) # 克隆环境 new_id = engine.clone_vm(vm_id, 'sandbox-copy')
06

安全隔离配置

# VMX 自动生成的安全配置 isolation.tools.hgfs.disable = "TRUE" # 禁用共享文件夹 isolation.tools.dnd.disable = "TRUE" # 禁用拖放 isolation.tools.copy.disable = "TRUE" # 禁用复制 usb.present = "FALSE" # 禁用USB ethernet0.connectionType = "hostonly" # 网络隔离
